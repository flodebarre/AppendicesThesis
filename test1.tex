	\renewcommand{\theequation}{\thechapter.A-\arabic{equation}}

	\renewcommand{\thefigure}{\thechapter.A-\arabic{figure}}
	\setcounter{figure}{0}
	%% Subsections
	\renewcommand{\thesubsection}{\thechapter.A-\arabic{subsection}}
}{%
	\end{subappendices}
}
