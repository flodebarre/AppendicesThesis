% Define name of the appendices (update with the name you want)
\def \appname {Annexe}
 
% Define a new environment for appendices
\newenvironment{miraineapp}{%

	\clearpage % Page break to start on new page (check also \cleardoublepage)
	
	\begin{subappendices}

	\section*{\appname} % Add unnumbered section called Annexe (update with the name you want)
	\markright{\sc \appname} % Change header (update it to match your style)
	\addcontentsline{toc}{section}{\appname} % Add to TOC

	% Update counters	
	%% Equations		
	\renewcommand{\theequation}{\thechapter.A-\arabic{equation}}
	\setcounter{equation}{0}  % reset counter
    %% Figures
	\renewcommand{\thefigure}{\thechapter.A-\arabic{figure}}
	\setcounter{figure}{0}
	%% Subsections
	\renewcommand{\thesubsection}{\thechapter.A-\arabic{subsection}}
}{%
	\end{subappendices}
}
